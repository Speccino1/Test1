<script>
    // Hardcoded login credentials for demonstration
    const validUsername = "user123";
    const validPassword = "pass123";

    // Login Function
    function login() {
        const username = document.getElementById('username').value;
        const password = document.getElementById('password').value;

        if (username === validUsername && password === validPassword) {
            // Hide login page and show configurator page
            document.getElementById('loginPage').classList.add('hidden');
            document.getElementById('configuratorPage').classList.remove('hidden');
        } else {
            // Show error message
            document.getElementById('errorMessage').classList.remove('hidden');
        }
    }

    // Generate Quote Function
    function generateQuote() {
        // Get values from Solar Installation
        const roofType = document.getElementById('roofType').options[document.getElementById('roofType').selectedIndex].text;
        const solarPanelBrand = document.getElementById('solarPanelBrand').options[document.getElementById('solarPanelBrand').selectedIndex].text;
        const solarPanelAmount = parseInt(document.getElementById('solarPanelAmount').value) || 0;
        const inverterBrand = document.getElementById('inverterBrand').options[document.getElementById('inverterBrand').selectedIndex].text;

        // Get values from Battery
        const batteryBrand = document.getElementById('batteryBrand').options[document.getElementById('batteryBrand').selectedIndex].text;
        const batterySize = parseInt(document.getElementById('batterySize').value) || 0;
        const smartMeterBrand = document.getElementById('smartMeterBrand').options[document.getElementById('smartMeterBrand').selectedIndex].text;

        // Get values from Chargers
        const chargerType = document.getElementById('chargerType').options[document.getElementById('chargerType').selectedIndex].text;

        // Calculate total price
        const roofTypeCost = parseFloat(document.getElementById('roofType').value);
        const solarPanelBrandCost = parseFloat(document.getElementById('solarPanelBrand').value) * solarPanelAmount;
        const inverterBrandCost = parseFloat(document.getElementById('inverterBrand').value);
        const batteryBrandCost = parseFloat(document.getElementById('batteryBrand').value) * batterySize;
        const smartMeterBrandCost = parseFloat(document.getElementById('smartMeterBrand').value);
        const chargerTypeCost = parseFloat(document.getElementById('chargerType').value);

        const totalPrice = roofTypeCost + solarPanelBrandCost + inverterBrandCost + batteryBrandCost + smartMeterBrandCost + chargerTypeCost;

        // Display values on the quote page
        document.getElementById('quoteRoofType').textContent = roofType;
        document.getElementById('quoteSolarPanelBrand').textContent = solarPanelBrand;
        document.getElementById('quoteSolarPanelAmount').textContent = solarPanelAmount;
        document.getElementById('quoteInverterBrand').textContent = inverterBrand;
        document.getElementById('quoteBatteryBrand').textContent = batteryBrand;
        document.getElementById('quoteBatterySize').textContent = batterySize;
        document.getElementById('quoteSmartMeterBrand').textContent = smartMeterBrand;
        document.getElementById('quoteChargerType').textContent = chargerType;
        document.getElementById('quoteTotalPrice').textContent = totalPrice.toFixed(2);

        // Hide configurator page and show quote page
        document.getElementById('configuratorPage').classList.add('hidden');
        document.getElementById('quotePage').classList.remove('hidden');
    }

    // Print Quote Function
    function printQuote() {
        window.print();
    }
</script>
